SMALL_COST = 3.25
MEDIUM_COST = 5.50
LARGE_COST = 7.15

EXTRA_TOPPING1 = 0.75
EXTRA_TOPPING2 = 1.35
EXTRA_TOPPING3 = 2.00
EXTRA_TOPPING4_OR_MORE = 2.50

DISCOUNT_RATE = 0.1

DELIVERY_CHARGE = 2.50

 DEF FUNCTION customerDetails():
	name = input("Enter name:")
    	address = input("Enter address:")
    	phoneNumber = input("Enter phone number:")
	return name, address, phoneNumber

 DEF FUNCTION validate_quantity_range(qty): 
    while qty < 1 or qty > 6:
        OUTPUT ("Not in range, please re-enter number of pizzas.")
        qty = int(input("Enter your number of pizzas from 1-6: "))
    return qty


DEF FUNCTION setBaseCost(size):
	if size == "Small":
        	bCost = SMALL_COST
    	elif size == "Medium":
        	bCost = MEDIUM_COST
    	elif size == "Large":
        	bCost = LARGE_COST
    	else:
        	print("Invalid size entered. Defaulting to Medium.")
        	bCost = MEDIUM_COST
    	return bCost
DEF FUNCTION setToppingsCost(nToppings):
    if nToppings == 1:
        toppingsCost = EXTRA_TOPPING1
    elif nToppings == 2:
        toppingsCost = EXTRA_TOPPING2
    elif nToppings == 3:
        toppingsCost = EXTRA_TOPPING3
    elif nToppings >= 4:
        toppingsCost = EXTRA_TOPPING4_OR_MORE
    else:
        toppingsCost = 0
    return toppingsCost

DEF FUNCTION orderSummary(name, address, phoneNumber, ttlCost, discount,delvryCost, finalTotal): 
    print("\n--- Order Summary ---")
    print(f"Customer Name: {name}")
    print(f"Address: {address}")
    print(f"Phone Number: {phoneNumber}")
    print(f"Subtotal: £{ttlCost:.2f}")
    print(f"Discount: £{discount:.2f}")
    print(f"Delivery Charge: £{delvryCost:.2f}")
    print(f"Total to Pay: £{finalTotal:.2f}")
    print("----------------------")

#Main program
custName, custAddress, custPhoneNumber = customerDetails()

quantity = int(input("Enter your number of pizzas from 1-6: "))
quantity = validate_quantity_range(quantity)

totalCost = 0 

for i in range(1, quantity + 1):
    OUTPUT(f"\nPizza {i}:")
    pizzaSize = input("Enter size (Small/Medium/Large): ").capitalize()
    baseCost =setBaseCost(pizzaSize)

    
    numToppings = int(input("Enter number of toppings: "))
    toppingsCost = setToppingsCost(numToppings)

    pizzaCost = baseCost + toppingsCost
    totalCost += pizzaCost

discount = 0
if totalCost >= 20:
    discountValue = totalCost * DISCOUNT_RATE

deliveryCost = 0
delivery = input("Delivery required? (Yes/No): ").capitalize()
if delivery == "Yes":
    deliveryCost = DELIVERY_CHARGE

finalPrice = totalCost + deliveryCost - discountValue

orderSummary(custName, custAddress, custPhoneNumber, totalCost, discountValue,deliveryCost, finalPrice)
